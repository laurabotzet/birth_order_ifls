add_likert_labels <- function(x) {
  val_labels(x) <- c("Disagree strongly" = 1, 
                  "Disagree a little" = 2, 
                  "Neither agree nor disagree" = 3,
                  "Agree a little" = 4,
                  "Agree strongly" = 5)
  x
}
